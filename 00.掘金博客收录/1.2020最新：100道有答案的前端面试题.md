## 写一个 mySetInterVal(fn, a, b),每次间隔 a,a+b,a+2b 的时间，然后写一个 myClear，停止上面的 mySetInterVal

```js
function mySetInterVal(fn, a, b) {
    this.a = a;
    this.b = b;
    this.time = 0;
    this.handle = -1;
    this.start = () => {
        this.handle = setTimeout(() => {
            fn();
            this.time++;
            this.start();
        }, this.a + this.time * this.b)
    };
    this.stop = () => {
        clearTimeout(this.handle);
        this.time = 0;
    }
}

function log() {
    console.log("hello world");
}

let a = new mySetInterVal(log, 1000, 1000);

a.start();

a.stop();
```

## 合并二维有序数组成一维有序数组，归并排序的思路

```js
function concatArr(arr) {
    let result = [];
    for (let i = 0; i < arr.length; i++) {
        result = result.concat(arr[i]);
    }
    return result.sort((n1, n2) => n1 - n2);
}

let arr1 = [1, 5, 7];
let arr2 = [2, 9, 6];
let arr3 = [4, 9, 10];
let arr = [arr1, arr2, arr3];

console.log(concatArr(arr));
```

## 多种方式实现斐波那契数列

> 递归方法

```js
function fib(n) {
    if (n < 0) {
        throw new Error("数字不能小于0!");
    } else if (n === 0) {
        return 0;
    } else if (n === 1) {
        return 1;
    } else {
        return fib(n - 1) + fib(n - 2);
    }
}
```
> 迭代

```js
function fib(n) {
    if (n < 0) {
        throw new Error("数字不能小于0!");
    } else if (n < 2) {
        return n;
    }
    let result = 0;
    let first = 0;
    let second = 1;
    for (let i = 2; i < n; i++) {
        result = first + second;
        second = first;
        first = result;
    }
    return result;
}
```
